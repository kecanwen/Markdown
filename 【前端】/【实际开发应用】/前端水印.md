//创建水印
createWaterMark(empName,userId) {
  let oWaterMark = document.querySelector('.waterMark');
  let mWaterMark = document.querySelector('.watermark');
  if(!oWaterMark && !mWaterMark) {
    let oDiv = document.createElement('div');
    oDiv.className = 'waterMark';
    let mDiv = document.createElement('div');
    mDiv.className = 'watermark';
    let aspan = document.createElement('span');
    let bspan = document.createElement('span');
    let cspan = document.createElement('span');
    let dspan = document.createElement('span');
    aspan.innerHTML = empName;
    bspan.innerHTML = userId;
    cspan.innerHTML = empName;
    dspan.innerHTML = userId;
    oDiv.appendChild(aspan);
    oDiv.appendChild(bspan);
    mDiv.appendChild(cspan);
    mDiv.appendChild(dspan);
    document.body.appendChild(oDiv);
    document.body.appendChild(mDiv);
  }
}
